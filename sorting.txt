*****************Bubble sort:*********************
Idea : Compare adjacent elements and swap in order

code:

*****************selection sort********************
idea : pick the smallest(from unsorted) and put it in the beginning
// Online C++ compiler to run C++ program online
#include <iostream>
#include<vector>
using namespace std;

int main() 
{
    vector<int> v={5,1,6,9,2};
    int n=v.size();
    for(int i=0;i<n-1;i++)
    {
        int minindex=i;
        for(int j=i+1;j<n;j++)
        {
            if(v[j]<v[minindex])
            {
                minindex=j;
            }
        }
        swap(v[i],v[minindex]);
    }
    cout<<"printing elements"<<endl;
    for(int i:v)
    {
        cout<<i<<endl;
    }

    return 0;
}

insertion sort :
IDEA : pick an element from unsorted part and place it correctly in sortedd part
// Online C++ compiler to run C++ program online
#include <iostream>
#include<vector>
using namespace std;

int main() 
{
    vector<int> v={5,1,6,9,2};
    int n=v.size();
    for(int i=1;i<n;i++)
    {
        int curr=v[i];
        int prev=i-1;
        while(prev>=0 && v[prev]>curr)
        {
            v[prev+1]=v[prev];
            prev=prev-1;
        }
        v[prev+1]=curr;
    }
    
    cout<<"printing elements"<<endl;
    for(int i:v)
    {
        cout<<i<<endl;
    }

    return 0;
}

*********************************merge sort:***********************
“I divide the array into halves recursively until each subarray has one element, then merge them back in sorted order. The merging step ensures the whole array is sorted, giving a time complexity of O(n log n).”
class Solution {
public:
    vector<int> sortArray(vector<int>& nums) {
        mergeSort(nums, 0, nums.size() - 1);
        return nums;
    }

private:
    void mergeSort(vector<int>& arr, int left, int right) {
        if (left >= right) return; // Base case: one element

        int mid = left + (right - left) / 2;

        // Sort left half
        mergeSort(arr, left, mid);

        // Sort right half
        mergeSort(arr, mid + 1, right);

        // Merge the two halves
        merge(arr, left, mid, right);
    }

    void merge(vector<int>& arr, int left, int mid, int right) {
        int n1 = mid - left + 1;
        int n2 = right - mid;

        // Temporary arrays
        vector<int> L(n1), R(n2);

        // Copy data
        for (int i = 0; i < n1; i++) L[i] = arr[left + i];
        for (int j = 0; j < n2; j++) R[j] = arr[mid + 1 + j];

        // Merge back into arr
        int i = 0, j = 0, k = left;
        while (i < n1 && j < n2) {
            if (L[i] <= R[j]) {
                arr[k++] = L[i++];
            } else {
                arr[k++] = R[j++];
            }
        }

        // Copy leftovers
        while (i < n1) arr[k++] = L[i++];
        while (j < n2) arr[k++] = R[j++];
    }
};

Quick Sort is a divide-and-conquer algorithm:

Pick a pivot (any element in the array, often the last or first).

Partition the array:

Put all elements smaller than the pivot to its left.

Put all elements greater than the pivot to its right.

Now the pivot is in its final sorted position.
